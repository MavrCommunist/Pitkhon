#Для проверки исполняемого файла нужно указать его и файл-заглушку
mypy <file> <another file> <directory>
#Для успешной проверки весь набор <> должен содержать и проверяемый файл и заглушку
mypy @<file with names of files to check>
#Можно записать полные имена файлов и директорй для проверки в текстовый файл построчно и скормить его mypy
